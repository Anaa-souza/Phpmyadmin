CREATE TABLE livros( 
id INT AUTO_INCREMET PRIMARY KEY,
titulo VARCHAR (65),
autorVARCHAR (50),
gênero VARCHAR (35),
Anopublicacao DATE 
);
CREATE TABLE usuario1 (
id INT AUTO_INCREMENT PRIMARY KEY,
nome VARCHAR (50),
email VARCHAR (50),
Tipousuario ENUM (professor, aluno, funcionario)
);

CREATE TABLE empréstimo (
id INT AUTO_INCREMENT PRIMARY KEY,
usuario_id INT NOT NULL,
livro_id INT NOT NULL,
Dataemprestimo DATE NOT NULL,
Datadevolucao DATE

);
